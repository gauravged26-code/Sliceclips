// components/AdvancedEditor.tsx
'use client';

import React, { useState, useRef, useEffect } from 'react';
import {
  Play,
  Pause,
  Download,
  Wand2,
  Sliders,
  Type,
  EyeOff,
  Share2,
  Scissors,
  FolderVideo
} from 'lucide-react';
import toast from 'react-hot-toast';
import { sliceVideoWasm, calculateViralityScore } from '@/lib/videoProcessing';
import ViralityScoreCard from './ViralityScoreCard';

interface WordTimestamp {
  word: string;
  start: number;
  end: number;
}

interface Clip {
  id: string;
  startTime: number;
  endTime: number;
  transcript: string;
  viralScore: number;
  viralityBreakdown: any;
}

interface EditorProps {
  videoUrl?: string;
  transcript?: string;
}

export default function AdvancedEditor({ videoUrl: initialVideoUrl = '', transcript: initialTranscript = '' }: EditorProps) {
  const videoRef = useRef<HTMLVideoElement>(null);
  
  // State Matrix
  const [videoSrc, setVideoSrc] = useState<string>(initialVideoUrl);
  const [videoFile, setVideoFile] = useState<File | null>(null);
  const [isProcessing, setIsProcessing] = useState(false);
  const [isPlaying, setIsPlaying] = useState(false);
  
  // AI, Transcription & Segment Tracks
  const [rawTranscript, setRawTranscript] = useState<string>(initialTranscript);
  const [words, setWords] = useState<WordTimestamp[]>([]);
  const [clips, setClips] = useState<Clip[]>([]);
  const [selectedClip, setSelectedClip] = useState<Clip | null>(null);

  // Caption Styling & Visibility Controls
  const [showCaptions, setShowCaptions] = useState(true);
  const [activeCaption, setActiveCaption] = useState('Process video to generate synchronized captions...');
  const [isLooping, setIsLooping] = useState(true);

  // Local Media Capture Handler
  const handleLocalVideoUpload = (e: React.ChangeEvent<HTMLInputElement>) => {
    if (e.target.files && e.target.files[0]) {
      const file = e.target.files[0];
      setVideoFile(file);
      setVideoSrc(URL.createObjectURL(file));
      toast.success('Media context loaded into memory workspace!');
    }
  };

  // OpenAI Whisper Engine Integration 
  const handleWhisperTranscription = async () => {
    if (!videoFile) return toast.error('Please upload or load a video file first.');
    setIsProcessing(true);
    toast.loading('Running AI Whisper transcription matching...', { id: 'whisper' });

    try {
      const formData = new FormData();
      formData.append('file', videoFile);

      const res = await fetch('/api/ai/transcribe', { method: 'POST', body: formData });
      const data = await res.json();

      if (!data.success) throw new Error(data.error || 'Transcription engine error');

      setRawTranscript(data.text);
      setWords(data.words || []);
      
      // Calculate algorithmic virality score over the full transcript duration
      const duration = videoRef.current?.duration || 30;
      const metrics = calculateViralityScore(data.text, duration);

      // Generate a dynamic initial viral slice from the timeline data
      const autoClip: Clip = {
        id: clip-${Date.now()},
        startTime: data.words[0]?.start || 0,
        endTime: data.words[data.words.length - 1]?.end || duration,
        transcript: data.text,
        viralScore: metrics.score,
        viralityBreakdown: metrics.breakdown
      };

      setClips([autoClip]);
      setSelectedClip(autoClip);
      toast.success('Whisper timestamp mapping completed!', { id: 'whisper' });
    } catch (err: any) {
      console.error(err);
      toast.error(Whisper processing failed: ${err.message}, { id: 'whisper' });
    } finally {
      setIsProcessing(false);
    }
  };

  // High-Performance WebAssembly FFmpeg Slicing
  const handleWasmDownload = async () => {
    if (!videoFile || !selectedClip) return toast.error('Missing clip segment parameters.');
    setIsProcessing(true);
    toast.loading('Compiling video slice via WebAssembly...', { id: 'wasm' });

    try {
      const slicedBlob = await sliceVideoWasm(videoFile, selectedClip.startTime, selectedClip.endTime);
      const url = URL.createObjectURL(slicedBlob);
      
      const a = document.createElement('a');
      a.href = url;
      a.download = slice_${selectedClip.id}.mp4;
      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);
      
      toast.success('WebAssembly compilation complete! Download started.', { id: 'wasm' });
    } catch (err: any) {
      console.error(err);
      toast.error(Wasm compilation failed: ${err.message}, { id: 'wasm' });
    } finally {
      setIsProcessing(false);
    }
  };

  // Share Link Loop Generation Strategy
  const handleGenerateShareableLoop = () => {
    if (!selectedClip) return toast.error('Select a clip target to loop.');
    const mockId = Math.random().toString(36).substring(7);
    const loopUrl = ${window.location.origin}/share/${mockId}?loop=true&start=${selectedClip.startTime}&end=${selectedClip.endTime};
    navigator.clipboard.writeText(loopUrl);
    toast.success('Shareable loop asset preview link copied to clipboard!');
  };

  const togglePlay = () => {
    if (!videoRef.current) return;
    if (isPlaying) {
      videoRef.current.pause();
    } else {
      videoRef.current.play();
    }
    setIsPlaying(!isPlaying);
  };

  return (
    <div className="bg-slate-950 border border-slate-800 rounded-xl p-6 grid grid-cols-1 lg:grid-cols-3 gap-6 text-slate-100 font-sans">
      
      {/* LEFT COLUMN: HIGH-PERFORMANCE VIDEO SANDBOX & LOOP CONTROLLER */}
      <div className="lg:col-span-2 space-y-4">
        <div className="relative aspect-video w-full bg-slate-900 rounded-lg overflow-hidden border border-slate-800 flex items-center justify-center group">
          {videoSrc ? (
            <>
              <video
                ref={videoRef}
                src={videoSrc}
                className="w-full h-full object-contain"
                loop={isLooping}
                onPlay={() => setIsPlaying(true)}
                onPause={() => setIsPlaying(false)}
                onTimeUpdate={() => {
                  if (!videoRef.current || words.length === 0) return;
                  const currentSec = videoRef.current.currentTime;
                  // Look up active transcript word context dynamically in real-time
                  const currentWordObj = words.find(w => currentSec >= w.start && currentSec <= w.end);
                  if (currentWordObj) setActiveCaption(currentWordObj.word);
                }}
              />
              {/* Dynamic Subtitle/Caption Burn Overlay Preview Box */}
              {showCaptions && (
                <div className="absolute bottom-10 left-1/2 -translate-x-1/2 px-4 py-1.5 bg-black/80 text-yellow-400 font-extrabold text-xl tracking-wide rounded border border-yellow-400/20 text-center pointer-events-none drop-shadow-md transition-all">
                  {activeCaption}
                </div>
              )}
            </>
          ) : (
            <div className="text-center p-8 border border-dashed border-slate-800 rounded-xl max-w-sm">
              <Scissors className="w-12 h-12 text-slate-700 mx-auto mb-3" />
              <p className="text-sm font-medium text-slate-400 mb-2">No active video canvas detected</p>
              <input 
                type="file" 
                accept="video/*" 
                onChange={handleLocalVideoUpload} 
                className="text-xs text-slate-500 file:mr-4 file:py-1 file:px-2 file:rounded file:border-0 file:text-xs file:font-semibold file:bg-indigo-600 file:text-white hover:file:bg-indigo-500 cursor-pointer" 
              />
            </div>
          )}
        </div>

        {/* Video Controls & Share Loop Strip */}
        <div className="flex flex-wrap items-center justify-between gap-4 bg-slate-900 p-3 rounded-lg border border-slate-800">
          <div className="flex items-center gap-2">
            <button 
              onClick={togglePlay} 
              disabled={!videoSrc}
              className="p-2 bg-indigo-600 hover:bg-indigo-500 disabled:bg-slate-800 text-white rounded-lg transition"
            >
              {isPlaying ? <Pause className="w-4 h-4" /> : <Play className="w-4 h-4" />}
            </button>
            
            <label className="flex items-center gap-2 cursor-pointer select-none text-xs font-medium text-slate-400 ml-2">
              <input 
                type="checkbox" 
                checked={isLooping} 
                onChange={(e) => setIsLooping(e.target.checked)} 
                className="rounded border-slate-700 bg-slate-950 accent-indigo-500 w-4 h-4" 
              />
              <span>Autoplay Loop Layer</span>
            </label>
          </div>

          <div className="flex items-center gap-2">
            <button 
              onClick={handleGenerateShareableLoop}
              disabled={!selectedClip}
              className="flex items-center gap-1.5 px-3 py-1.5 bg-slate-800 hover:bg-slate-700 text-slate-200 border border-slate-700 rounded-lg text-xs font-medium transition"
            >
              <Share2 className="w-3.5 h-3.5 text-indigo-400" />
              <span>Generate Share Loop</span>
            </button>
            
            <button 
              onClick={handleWasmDownload}
              disabled={isProcessing || !selectedClip}
              className="flex items-center gap-1.5 px-3 py-1.5 bg-indigo-600 hover:bg-indigo-500 disabled:bg-slate-800 text-white rounded-lg text-xs font-medium transition"
            >
              <Download className="w-3.5 h-3.5" />
              <span>Wasm Export</span>
            </button>
          </div>
        </div>
      </div>

      {/* RIGHT COLUMN: AI ORCHESTRATION, TRANSCRIPTION MATRIX, CAPTION ELIMINATOR */}
      <div className="space-y-4">
        
        {/* OpenAI Whisper Trigger */}
        <div className="bg-slate-900 border border-slate-800 rounded-lg p-4 space-y-3">
          <h3 className="font-bold text-xs tracking-wider text-slate-400 uppercase">AI Orchestration</h3>
          <button
            onClick={handleWhisperTranscription}
            disabled={isProcessing || !videoFile}
            className="w-full flex items-center justify-center gap-2 bg-indigo-600 hover:bg-indigo-500 disabled:bg-slate-800 text-white py-2 px-4 rounded-lg text-sm font-medium transition"
          >
            <Wand2 className="w-4 h-4" />
            <span>{isProcessing ? 'Whisper Engine Running...' : 'Run Whisper Engine'}</span>
          </button>
        </div>

        {/* Caption Removal and Overlay Isolator */}
        <div className="bg-slate-900 border border-slate-800 rounded-lg p-4 space-y-3">
          <div className="flex items-center justify-between">
            <h3 className="font-bold text-xs tracking-wider text-slate-400 uppercase">Caption Track Controls</h3>
            <span className={`text-[10px] px-1.5 py-0.5 rounded font-medium ${showCaptions ? 'bg-emerald-500/10 text-emerald-400' : 'bg-rose-500/10 text-rose-400'}`}>
              {showCaptions ? 'Active Overlay' : 'Muted/Removed'}
            </span>
          </div>

          <div className="grid grid-cols-2 gap-2">
            <button
              onClick={() => {
                setShowCaptions(true);
                toast.success("AI Caption generation track isolated!");
              }}
              className="flex items-center justify-center gap-1.5 py-2 px-3 bg-slate-800 hover:bg-slate-750 border border-slate-700 rounded-md text-xs font-medium transition"
            >
              <Type className="w-3.5 h-3.5 text-indigo-400" />
              <span>Isolate Captions</span>
            </button>
            
            <button
              onClick={() => {
                setShowCaptions(false);
                setActiveCaption('');
                toast.success("Burned subtitles suppressed cleanly!");
              }}
              className="flex items-center justify-center gap-1.5 py-2 px-3 bg-rose-950/10 hover:bg-rose-950/20 border border-rose-900/30 text-rose-400 rounded-md text-xs font-medium transition"
            >
              <EyeOff className="w-3.5 h-3.5" />
              <span>Remove Captions</span>
            </button>
          </div>
        </div>

        {/* Dynamic Multi-Signal Virality Output */}
        {selectedClip && (
          <div className="bg-slate-900 border border-slate-800 rounded-lg p-4">
            <h3 className="font-bold text-xs tracking-wider text-slate-400 uppercase mb-3">AI Virality Index</h3>
            <ViralityScoreCard score={selectedClip.viralScore} breakdown={selectedClip.viralityBreakdown} />
          </div>
        )}

        {/* Interactive Word Mapping Deep Timeline Container */}
        <div className="bg-slate-900 border border-slate-800 rounded-lg p-4 flex flex-col h-[180px]">
          <h3 className="font-bold text-xs tracking-wider text-slate-400 uppercase">Interactive Timeline Matrix</h3>
          <div className="flex-1 overflow-y-auto text-xs text-slate-400 space-x-1 space-y-1 mt-2 pr-1 scrollbar-thin">
            {words.length > 0 ? (
              words.map((w, i) => (
                <span
                  key={i}
                  className="inline-block cursor-pointer px-1 py-0.5 rounded bg-slate-950 border border-slate-800 hover:bg-indigo-600 hover:text-white transition text-slate-300"
                  onClick={() => {
                    if (videoRef.current) {
                      videoRef.current.currentTime = w.start;
                    }
                  }}
                >
                  {w.word}
                </span>
              ))
            ) : (
              <div className="h-full flex flex-col items-center justify-center text-slate-600 text-center italic p-4">
                <FolderVideo className="w-6 h-6 mb-1 text-slate-700" />
                <span>No track segments mapped. Run Whisper AI to bind timelines.</span>
              </div>
            )}
          </div>
        </div>

      </div>
    </div>
  );
}
